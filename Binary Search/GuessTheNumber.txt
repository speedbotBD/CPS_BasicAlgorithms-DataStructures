Problem link-> https://leetcode.com/problems/guess-number-higher-or-lower/

class Solution {
public:
    int guessNumber(int n) {
        
        int l = 1, r = n;
        long long ans = -1;

        while(l<=r){
          
          int mid = l + (r - l)/2;
          
          int a = guess(mid);

          if(a == 0) return mid;
          else if(a == 1){
            l = mid + 1;
          }else{
            r = mid - 1;
          }

        }

          return 1;
    }
    
};
